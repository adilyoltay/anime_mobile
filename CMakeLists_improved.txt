cmake_minimum_required(VERSION 3.16)
project(ImprovedJsonConverter)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Include nlohmann/json (header-only library)
include(FetchContent)
FetchContent_Declare(
    json
    GIT_REPOSITORY https://github.com/nlohmann/json.git
    GIT_TAG v3.11.2
)
FetchContent_MakeAvailable(json)

# Improved converter executable (minimal dependencies)
add_executable(improved_converter
    improved_json_converter.cpp
)

target_link_libraries(improved_converter
    nlohmann_json::nlohmann_json
)

# Install targets
install(TARGETS improved_converter
    RUNTIME DESTINATION bin
)

install(FILES 
    bouncing_ball.json
    DESTINATION share/examples
)

# Print configuration info
message(STATUS "Improved JSON Converter - Minimal Dependencies with Proper RIV Structure")
