cmake_minimum_required(VERSION 3.16)
project(SimpleJsonConverter)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Include nlohmann/json (header-only library)
include(FetchContent)
FetchContent_Declare(
    json
    GIT_REPOSITORY https://github.com/nlohmann/json.git
    GIT_TAG v3.11.2
)
FetchContent_MakeAvailable(json)

# Simple converter executable (no Rive runtime dependency)
add_executable(simple_json_converter
    simple_json_converter.cpp
)

target_link_libraries(simple_json_converter
    nlohmann_json::nlohmann_json
)

# Install targets
install(TARGETS simple_json_converter
    RUNTIME DESTINATION bin
)

# Print configuration info
message(STATUS "Simple JSON Converter - No Rive Runtime Dependencies")
